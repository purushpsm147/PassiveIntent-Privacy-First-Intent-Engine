Simple API. Strong defaults.

Drop PassiveIntent into any SPA and react instantly to behavioral signals with full client-side control. 

import { IntentManager } from '@passiveintent/core';

const intent = new IntentManager({ storageKey: 'passive-intent' });

intent.track('/checkout/payment');

intent.on('dwell_time_anomaly', ({ zScore }) => {
  if (zScore > 2.0) showTooltip('Need help finishing this step?');
});
